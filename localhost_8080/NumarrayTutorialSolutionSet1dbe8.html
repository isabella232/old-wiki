Exercise 5:

 >>> from numarray import *
 >>> y, x = indices((500,500)) # ints, next statements makes floats
 >>> y = y - 250.
 >>> x = x - 250.
 >>> im = sin(x/pi)*exp(-(x**2+y**)/2500.)
 >>> from numarray.fft import *
 >>> fim = fft2d(im)
 >>> import numdisplay
 >>> numdisplay.display(abs(fim))
