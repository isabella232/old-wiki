   1 """Create plots of signals generated by chirp() and sweep_poly()."""
   2 
   3 import numpy as np
   4 from scipy.signal.waveforms import chirp, sweep_poly
   5 from numpy import poly1d
   6 
   7 from pylab import figure, plot, show, xlabel, ylabel, subplot, grid, title, \
   8                     yscale, savefig, clf
   9 
  10 FIG_SIZE = (7.5, 3.75)
  11 
  12 def make_linear(f0, t1, f1, filename=None, fig_size=FIG_SIZE):
  13     t = np.linspace(0, t1, 5001)
  14     w = chirp(t, f0=f0, f1=f1, t1=t1, method='linear')
  15 
  16     figure(1, figsize=fig_size)
  17     clf()
  18 
  19     subplot(2,1,1)
  20     plot(t, w)
  21     tstr = "Linear Chirp, f(0)=%g, f(%g)=%g" % (f0, t1, f1)
  22     title(tstr)
  23 
  24     subplot(2,1,2)
  25     plot(t, f0 + (f1-f0)*t/t1, 'r')
  26     grid(True)
  27     ylabel('Frequency (Hz)')
  28     xlabel('time (sec)')
  29     if filename is None:
  30         show()
  31     else:
  32         savefig(filename)
  33 
  34 def make_quadratic(f0, t1, f1, filename=None, fig_size=FIG_SIZE):
  35     t = np.linspace(0, t1, 5001)

  37 
  38     figure(1, figsize=fig_size)
  39     clf()
  40 
  41     subplot(2,1,1)
  42     plot(t, w)

  48     grid(True)
  49     ylabel('Frequency (Hz)')
  50     xlabel('time (sec)')
  51     if filename is None:
  52         show()

  54         savefig(filename)

  56 def make_quadratic_v0false(f0, t1, f1, filename=None, fig_size=FIG_SIZE):
  57     t = np.linspace(0, t1, 5001)
  58     w = chirp(t, f0=f0, f1=f1, t1=t1, method='quadratic', vertex_zero=False)
  59 
  60     figure(1, figsize=fig_size)
  61     clf()
  62 
  63     subplot(2,1,1)
  64     plot(t, w)
  65     tstr = "Quadratic Chirp, f(0)=%g, f(%g)=%g (vertex_zero=False)" % (f0, t1, f1)
  66     title(tstr)
  67 
  68     subplot(2,1,2)
  69     plot(t, f1 - (f1-f0)*(t1-t)**2/t1**2, 'r')
  70     grid(True)
  71     ylabel('Frequency (Hz)')
  72     xlabel('time (sec)')
  73     if filename is None:
  74         show()
  75     else:
  76         savefig(filename)

 123 def make_sweep_poly(filename=None, fig_size=FIG_SIZE):
 124     p = poly1d([0.05, -0.75, 2.5, 5.0])
 125      
 126     t = np.linspace(0, t1, 5001)
 127     w = sweep_poly(t, p)
 128 
 129     figure(1, figsize=fig_size)
 130     clf()
 131     
 132     subplot(2,1,1)
 133     plot(t, w)
 134     tstr = "Sweep Poly, $f(t) = 0.05t^3 - 0.75t^2 + 2.5t + 5$"
 135     title(tstr)
 136 
 137     subplot(2,1,2)
 138     plot(t, p(t), 'r')
 139     grid(True)
 140     ylabel('Frequency (Hz)')
 141     xlabel('time (sec)')
 142     if filename is None:
 143         show()
 144     else:
 145         savefig(filename)
